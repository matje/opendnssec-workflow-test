	--config <config>          aka -c

## commands

	setup
	update

	zone add
		--zone <zone>                     aka -z
		[--policy <policy>]               aka -p
		[--signerconf <signerconf.xml>]   aka -s
		[--input <input>]                 aka -i
		[--output <output>]               aka -o

	zone delete
		--zone <zone> | --all             aka -z / -a

	zone list

	repository list

	policy export
		--policy <policy> | --all         aka -p / -a

	policy list
